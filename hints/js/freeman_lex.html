That's almost all. Just add function statement and return res: 

function splitPairs(text: string): string[] {

    if (text.length % 2)
        text += "_";
    
    let res: string[] = [];
    while (text) {
        res.push(text.slice(0, 2));
        text = text.slice(2);
    }             
    return res;
}

Iterate over the text, taking each time next two characters. 

function splitPairs(text: string): string[] {
    
    let result: string[] = [];
    for(let i: number = 0; i < text.length; i += 2){
        result.push((text + '_').slice(i, i + 2));
    }
    return result;
}
